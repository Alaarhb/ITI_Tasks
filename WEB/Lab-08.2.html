const calculate = (operator, ...numbers) => {
    let result;
    switch(operator) {
        case '+':
            result = numbers.reduce((sum, num) => sum + num, 0);
            break;
        case '-':
            result = numbers.reduce((diff, num) => diff - num);
            break;
        case '*':
            result = numbers.reduce((prod, num) => prod * num, 1);
            break;
        case '/':
            result = numbers.reduce((quot, num) => quot / num);
            break;
    }
    console.log(`result of ${operator} operation for ${numbers.join(',')} is ${result}`);
};
